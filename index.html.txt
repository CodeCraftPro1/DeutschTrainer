  function shuffleArray(arr) {
    for(let i = arr.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [arr[i], arr[j]] = [arr[j], arr[i]];
    }
  }

  function flashBackground(colorClass) {
    screen.classList.add(colorClass === 'correct' ? 'correct-flash' : 'wrong-flash');
    setTimeout(() => {
      screen.classList.remove('correct-flash');

  function showQuestion() {
    emojiEl.textContent = '';
    optionsContainer.innerHTML = '';

    if(currentIndex >= words.length) {
      alert(`Challenge Complete! Well done.`);
      currentIndex = 0;
    }

    const word = words[currentIndex];
    englishWordEl.textContent = word.english;

    let options = [word.german];

    while (options.length < 3) {
      let idx = Math.floor(Math.random() * words.length);
      let candidate = words[idx].german;
      if (!options.includes(candidate)) {
        options.push(candidate);
      }
    }

    shuffleArray(options);

    options.forEach(opt => {
      const btn = document.createElement('div');
      btn.textContent = opt;
      btn.classList.add('option');
      btn.addEventListener('click', () => handleAnswer(btn, word.german));
      optionsContainer.appendChild(btn);
    });
  }

  function handleAnswer(selectedBtn, correctAnswer) {
    const allOptions = document.querySelectorAll('.option');
    allOptions.forEach(btn => btn.classList.add('disabled'));

    if (selectedBtn.textContent === correctAnswer) {
      selectedBtn.classList.add('correct');
      emojiEl.textContent = "‚úÖ‚ú®";
      flashBackground('correct');
      correctSound.play();
    } else {
      selectedBtn.classList.add('wrong');
      emojiEl.textContent = "‚ùå";
      flashBackground('wrong');
      wrongSound.play();
      allOptions.forEach(btn => {
        if (btn.textContent === correctAnswer) btn.classList.add('correct');
      });
    }

    setTimeout(() => {
      currentIndex++;
      showQuestion();
    }, 1200);
  }
